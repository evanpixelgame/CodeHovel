import React from "react";
import HeaderControls from "./controls/HeaderControls";
import FooterControls from "./controls/FooterControls";
import PresetSelector from "./subcomponents/PresetSelector";

const renderControls = (selectedSection, themeProperties, handleInputChangeWithSetter) => {
  switch (selectedSection) {
    case "header":
      return (
        <HeaderControls
          themeProperties={themeProperties}
          handleInputChange={handleInputChangeWithSetter}
        />
      );
    case "footer":
      return (
        <FooterControls
          themeProperties={themeProperties}
          handleInputChange={handleInputChangeWithSetter}
        />
      );
    case "presetThemes":
      return <PresetSelector />;
    default:
      return null;
  }
};

export default renderControls;
